class UploadHandler:
    """é€šç”¨ä¸Šä¼ å¤„ç†å™¨"""
    
    def __init__(self, upload_dir='uploads/', max_size=5*1024*1024, allowed_types=None):
        self.upload_dir = upload_dir
        self.max_size = max_size
        self.allowed_types = allowed_types or ['image/jpeg', 'image/png', 'image/gif', 'image/webp']
    
    def validate_file(self, uploaded_file):
        """éªŒè¯æ–‡ä»¶"""
        # æ£€æŸ¥æ–‡ä»¶å¤§å°
        if uploaded_file.size > self.max_size:
            raise ValueError(f'æ–‡ä»¶å¤§å°ä¸èƒ½è¶…è¿‡{self.max_size/1024/1024}MB')
        
        # æ£€æŸ¥æ–‡ä»¶ç±»å‹
        if uploaded_file.content_type not in self.allowed_types:
            raise ValueError('ä¸æ”¯æŒçš„æ–‡ä»¶ç±»å‹')
        
        return True
    
    def save_file(self, uploaded_file, sub_dir=''):
        """ä¿å­˜æ–‡ä»¶"""
        # éªŒè¯æ–‡ä»¶
        self.validate_file(uploaded_file)
        
        # ç”Ÿæˆæ–‡ä»¶å
        file_extension = os.path.splitext(uploaded_file.name)[1]
        unique_filename = f"{uuid.uuid4().hex}{file_extension}"
        file_path = os.path.join(self.upload_dir, sub_dir, unique_filename)
        
        # ä¿å­˜æ–‡ä»¶
        saved_path = default_storage.save(file_path, ContentFile(uploaded_file.read()))
        
        return {
            'url': default_storage.url(saved_path),
            'filename': unique_filename,
            'path': saved_path,
            'size': uploaded_file.size
        }
    
    def process_image(self, uploaded_file, sub_dir='', resize=None, quality=85):
        """å¤„ç†å›¾ç‰‡"""
        # ä¿å­˜åŸå§‹æ–‡ä»¶

                    # ä¿å­˜å¤„ç†åçš„å›¾ç‰‡
