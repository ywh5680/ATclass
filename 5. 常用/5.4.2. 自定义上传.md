## 自定义上传

Django提供了灵活的文件上传机制，允许开发者根据具体需求自定义上传功能。通过自定义上传，可以实现拖拽上传、图片裁剪、进度显示、多文件上传等高级功能，大大提升用户体验。

### 自定义上传基础

#### 1. 基本自定义上传视图

```python
# views.py
from django.http import JsonResponse
from django.views.decorators.csrf import csrf_exempt
from django.views.decorators.http import require_http_methods
from django.core.files.storage import default_storage
from django.core.files.base import ContentFile
import uuid
import os

@csrf_exempt
@require_http_methods(["POST"])
def custom_upload(request):
    """自定义文件上传"""
    try:
        if 'file' not in request.FILES:
            return JsonResponse({
                'success': False,
                'error': '没有选择文件'
            })
        
        uploaded_file = request.FILES['file']
        
        # 验证文件类型
        allowed_types = ['image/jpeg', 'image/png', 'image/gif', 'image/webp']
        if uploaded_file.content_type not in allowed_types:
            return JsonResponse({
                'success': False,
                'error': '不支持的文件类型'
            })
        
        # 验证文件大小（5MB）
        if uploaded_file.size > 5 * 1024 * 1024:
            return JsonResponse({
                'success': False,
                'error': '文件大小不能超过5MB'
            })
        
        # 生成唯一文件名
        file_extension = os.path.splitext(uploaded_file.name)[1]
        unique_filename = f"{uuid.uuid4().hex}{file_extension}"
        file_path = f"uploads/{unique_filename}"
        
        # 保存文件
        saved_path = default_storage.save(file_path, ContentFile(uploaded_file.read()))
        
        return JsonResponse({
            'success': True,
            'url': default_storage.url(saved_path),
            'filename': unique_filename,
            'original_name': uploaded_file.name,
            'size': uploaded_file.size
        })
        
    except Exception as e:
        return JsonResponse({
            'success': False,
            'error': str(e)
        })
```

#### 2. 带进度显示的上传

```python
# views.py
from django.http import JsonResponse, StreamingHttpResponse
from django.views.decorators.csrf import csrf_exempt
import json

@csrf_exempt
def upload_with_progress(request):
    """带进度显示的文件上传"""
    if request.method == 'POST':
        try:
            uploaded_file = request.FILES.get('file')
            if not uploaded_file:
                return JsonResponse({'success': False, 'error': '没有文件'})
            
            # 获取文件信息
            file_size = uploaded_file.size
            chunk_size = 8192  # 8KB chunks
            uploaded_size = 0
            
            # 生成文件名
            unique_filename = f"{uuid.uuid4().hex}{os.path.splitext(uploaded_file.name)[1]}"
            file_path = f"uploads/{unique_filename}"
            
            # 分块读取和保存
            with default_storage.open(file_path, 'wb') as destination:
                for chunk in uploaded_file.chunks(chunk_size):
                    destination.write(chunk)
                    uploaded_size += len(chunk)
                    
                    # 计算进度
                    progress = (uploaded_size / file_size) * 100
            
            return JsonResponse({
                'success': True,
                'url': default_storage.url(file_path),
                'filename': unique_filename,
                'progress': 100
            })
            
        except Exception as e:
            return JsonResponse({'success': False, 'error': str(e)})
    
    return JsonResponse({'success': False, 'error': '只支持POST请求'})
```

### 拖拽上传

#### 1. 拖拽上传视图

```python
# views.py
from django.http import JsonResponse
from django.views.decorators.csrf import csrf_exempt
from django.views.decorators.http import require_http_methods
import json

@csrf_exempt
@require_http_methods(["POST"])
def drag_drop_upload(request):
    """拖拽上传处理"""
    try:
        # 处理多个文件
        uploaded_files = request.FILES.getlist('files')
        
        if not uploaded_files:
            return JsonResponse({
                'success': False,
                'error': '没有选择文件'
            })
        
        results = []
        
        for uploaded_file in uploaded_files:
            # 验证文件
            if not validate_file(uploaded_file):
                results.append({
                    'filename': uploaded_file.name,
                    'success': False,
                    'error': '文件验证失败'
                })
                continue
            
            # 保存文件
            try:
                unique_filename = f"{uuid.uuid4().hex}{os.path.splitext(uploaded_file.name)[1]}"
                file_path = f"uploads/{unique_filename}"
                
                saved_path = default_storage.save(file_path, ContentFile(uploaded_file.read()))
                
                results.append({
                    'filename': uploaded_file.name,
                    'success': True,
                    'url': default_storage.url(saved_path),
                    'size': uploaded_file.size
                })
                
            except Exception as e:
                results.append({
                    'filename': uploaded_file.name,
                    'success': False,
                    'error': str(e)
                })
        
        return JsonResponse({
            'success': True,
            'results': results
        })
        
    except Exception as e:
        return JsonResponse({
            'success': False,
            'error': str(e)
        })

def validate_file(uploaded_file):
    """验证文件"""
    # 检查文件类型
    allowed_types = ['image/jpeg', 'image/png', 'image/gif', 'image/webp']
    if uploaded_file.content_type not in allowed_types:
        return False
    
    # 检查文件大小
    if uploaded_file.size > 5 * 1024 * 1024:
        return False
    
    return True
```

    constructor() {
        this.uploadArea = document.getElementById('uploadArea');
        this.fileInput = document.getElementById('fileInput');
        this.fileList = document.getElementById('fileList');
        this.files = [];
        
        this.initEventListeners();

    initEventListeners() {
        // 拖拽事件
        this.uploadArea.addEventListener('dragover', (e) => {
            e.preventDefault();
            this.uploadArea.classList.add('dragover');
        });
        
        this.uploadArea.addEventListener('dragleave', (e) => {
            e.preventDefault();
            this.uploadArea.classList.remove('dragover');
        });
        
        this.uploadArea.addEventListener('drop', (e) => {
            e.preventDefault();
            this.uploadArea.classList.remove('dragover');
            
            const files = Array.from(e.dataTransfer.files);
            this.handleFiles(files);
        });
        
        // 点击选择文件
        this.uploadArea.addEventListener('click', () => {
            this.fileInput.click();
        });
        
        this.fileInput.addEventListener('change', (e) => {
            const files = Array.from(e.target.files);
            this.handleFiles(files);
        });
    }
    
    handleFiles(files) {
        files.forEach(file => {
            if (file.type.startsWith('image/')) {
                this.addFileToList(file);
                this.uploadFile(file);
            }
        });
    }
    
    addFileToList(file) {
        const fileItem = document.createElement('div');
        fileItem.className = 'file-item';
        fileItem.innerHTML = `
            <img src="${URL.createObjectURL(file)}" alt="${file.name}">
            <div class="file-info">
                <div class="file-name">${file.name}</div>
                <div class="file-size">${this.formatFileSize(file.size)}</div>
                <div class="upload-progress">
                    <div class="progress-bar"></div>
                </div>
            </div>
            <div class="upload-status">上传中...</div>
        `;
        
        this.fileList.appendChild(fileItem);
        file.itemElement = fileItem;
    }
    
    async uploadFile(file) {
        const formData = new FormData();
        formData.append('file', file);
        
        try {
            const response = await fetch('/upload/', {
                method: 'POST',
                body: formData
            });
            
            const result = await response.json();
            
            if (result.success) {
                this.updateFileStatus(file, 'success', '上传成功');
                this.updateProgress(file, 100);
            } else {
                this.updateFileStatus(file, 'error', result.error);
            }
            
        } catch (error) {
            this.updateFileStatus(file, 'error', '上传失败');
        }

        const k = 1024;
        const sizes = ['Bytes', 'KB', 'MB', 'GB'];
        const i = Math.floor(Math.log(bytes) / Math.log(k));
        return parseFloat((bytes / Math.pow(k, i)).toFixed(2)) + ' ' + sizes[i];
    }

### 图片裁剪上传

#### 1. 裁剪上传视图

```python
# views.py
from django.http import JsonResponse
from django.views.decorators.csrf import csrf_exempt
from PIL import Image
import base64
import io

@csrf_exempt
def crop_upload(request):
    """图片裁剪上传"""
    if request.method == 'POST':
        try:
            # 获取裁剪数据
            data = json.loads(request.body)
            image_data = data.get('image')
            crop_data = data.get('crop')
            
            if not image_data or not crop_data:
                return JsonResponse({
                    'success': False,
                    'error': '参数不完整'
                })
            
            # 解码Base64图片数据
            image_data = image_data.split(',')[1]  # 移除data:image/jpeg;base64,前缀
            image_bytes = base64.b64decode(image_data)
            
            # 打开图片
            with Image.open(io.BytesIO(image_bytes)) as img:
                # 裁剪图片
                cropped_img = img.crop((
                    crop_data['x'],
                    crop_data['y'],
                    crop_data['x'] + crop_data['width'],
                    crop_data['y'] + crop_data['height']
                ))
                
                # 调整尺寸（如果需要）
                if crop_data.get('target_width') and crop_data.get('target_height'):
                    cropped_img = cropped_img.resize(
                        (crop_data['target_width'], crop_data['target_height']),
                        Image.Resampling.LANCZOS
                    )
                
                # 保存裁剪后的图片
                output = io.BytesIO()
                cropped_img.save(output, format='JPEG', quality=85, optimize=True)
                output.seek(0)
                
                # 生成文件名并保存
                unique_filename = f"{uuid.uuid4().hex}.jpg"
                file_path = f"uploads/cropped/{unique_filename}"
                
                saved_path = default_storage.save(file_path, ContentFile(output.getvalue()))
                
                return JsonResponse({
                    'success': True,
                    'url': default_storage.url(saved_path),
                    'filename': unique_filename
                })
                
        except Exception as e:
            return JsonResponse({
                'success': False,
                'error': str(e)
            })
    
    return JsonResponse({'success': False, 'error': '只支持POST请求'})
```

            uploaded_files = request.FILES.getlist('files')
            
            if not uploaded_files:
                return JsonResponse({
                    'success': False,
                    'error': '没有选择文件'
                })
            
            results = []
            
            # 使用线程池处理多个文件
            with ThreadPoolExecutor(max_workers=3) as executor:
                futures = []
                
                for uploaded_file in uploaded_files:
                    future = executor.submit(process_single_file, uploaded_file)

                # 收集结果
                for future in futures:

                    results.append(result)

                'success': True,
                'results': results
            })
            
        except Exception as e:
            return JsonResponse({
                'success': False,
                'error': str(e)

### 上传组件封装

#### 1. 通用上传组件

```python
# upload_utils.py
from django.core.files.storage import default_storage
from django.core.files.base import ContentFile
from PIL import Image
import uuid
import os

class UploadHandler:
    """通用上传处理器"""
    
    def __init__(self, upload_dir='uploads/', max_size=5*1024*1024, allowed_types=None):
        self.upload_dir = upload_dir
        self.max_size = max_size
        self.allowed_types = allowed_types or ['image/jpeg', 'image/png', 'image/gif', 'image/webp']
    
    def validate_file(self, uploaded_file):
        """验证文件"""
        # 检查文件大小
        if uploaded_file.size > self.max_size:
            raise ValueError(f'文件大小不能超过{self.max_size/1024/1024}MB')
        
        # 检查文件类型
        if uploaded_file.content_type not in self.allowed_types:
            raise ValueError('不支持的文件类型')
        
        return True
    
    def save_file(self, uploaded_file, sub_dir=''):
        """保存文件"""
        # 验证文件
        self.validate_file(uploaded_file)
        
        # 生成文件名
        file_extension = os.path.splitext(uploaded_file.name)[1]
        unique_filename = f"{uuid.uuid4().hex}{file_extension}"
        file_path = os.path.join(self.upload_dir, sub_dir, unique_filename)
        
        # 保存文件
        saved_path = default_storage.save(file_path, ContentFile(uploaded_file.read()))
        
        return {
            'url': default_storage.url(saved_path),
            'filename': unique_filename,
            'path': saved_path,
            'size': uploaded_file.size
        }
    
    def process_image(self, uploaded_file, sub_dir='', resize=None, quality=85):
        """处理图片"""
        # 保存原始文件
        result = self.save_file(uploaded_file, sub_dir)
        
        # 处理图片
        if resize:
            try:
                with Image.open(default_storage.path(result['path'])) as img:
                    # 调整尺寸
                    img.thumbnail(resize, Image.Resampling.LANCZOS)
                    
                    # 保存处理后的图片
                    img.save(default_storage.path(result['path']), 'JPEG', quality=quality, optimize=True)
                    
            except Exception as e:
                print(f"处理图片失败: {e}")
        
        return result
```
