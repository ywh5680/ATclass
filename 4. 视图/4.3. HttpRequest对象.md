## HttpRequest对象

Django的HttpRequest对象包含了关于HTTP请求的所有信息，包括请求头、请求体、查询参数、文件上传等。理解HttpRequest对象的结构和使用方法对于构建功能完整的Web应用至关重要。

### 什么是HttpRequest对象

HttpRequest对象是Django自动创建的，包含了当前HTTP请求的所有信息。当Django接收到HTTP请求时，会创建一个HttpRequest实例，并将其作为第一个参数传递给视图函数。

#### 基本概念

```python
from django.shortcuts import render
from django.http import HttpResponse, JsonResponse

def example_view(request):
    """示例视图，展示HttpRequest对象的使用"""
    # request是HttpRequest实例
    print(f"请求方法: {request.method}")
    print(f"请求路径: {request.path}")
    print(f"请求URL: {request.build_absolute_uri()}")
    print(f"用户代理: {request.META.get('HTTP_USER_AGENT')}")
    
    return HttpResponse("请求信息已打印到控制台")
```

def search_view(request):
    """搜索视图，处理GET参数"""
    
    # 获取单个GET参数
    query = request.GET.get('q', '')  # 搜索查询，默认为空字符串
    page = request.GET.get('page', '1')  # 页码，默认为1
    
    # 获取多个同名参数
    categories = request.GET.getlist('category')  # 分类列表
    
    # 检查参数是否存在
    has_sort = 'sort' in request.GET
    has_filter = 'filter' in request.GET
    
    # 获取所有GET参数
    all_params = request.GET.dict()  # 转换为字典
    
    # 构建查询
    articles = Article.objects.all()
    
    if query:
        articles = articles.filter(title__icontains=query)
    
    if categories:
        articles = articles.filter(category__slug__in=categories)
    
    # 分页
    try:
        page = int(page)
        if page < 1:
            page = 1
    except ValueError:
        page = 1
    
    context = {
        'query': query,
        'page': page,
        'categories': categories,
        'has_sort': has_sort,
        'has_filter': has_filter,
        'all_params': all_params,
        'articles': articles,
    }
    
    return render(request, 'search.html', context)

@csrf_exempt  # 注意：生产环境不要使用
def article_create_view(request):
    """创建文章视图，处理POST参数"""
    
    if request.method != 'POST':
        return JsonResponse({'error': '只支持POST请求'}, status=405)
    
    # 获取POST参数
    title = request.POST.get('title')
    content = request.POST.get('content')
    category_id = request.POST.get('category')
    
    # 获取多个同名参数
    tags = request.POST.getlist('tags')
    
    # 验证必填字段
    if not title or not content or not category_id:
        return JsonResponse({
            'error': '请填写所有必填字段',
            'missing_fields': []
        }, status=400)
    
    missing_fields = []
    if not title:
        missing_fields.append('title')
    if not content:
        missing_fields.append('content')
    if not category_id:
        missing_fields.append('category')
    
    if missing_fields:
        return JsonResponse({
            'error': '请填写所有必填字段',
            'missing_fields': missing_fields
        }, status=400)
    
    try:
        # 创建文章
        category = Category.objects.get(pk=category_id)
        article = Article.objects.create(
            title=title,
            content=content,
            category=category,
            author=request.user
        )
        
        # 添加标签
        if tags:
            article.tags.set(tags)
        
        return JsonResponse({
            'success': True,
            'article_id': article.id,
            'message': '文章创建成功'
        })
        
    except Category.DoesNotExist:
        return JsonResponse({
            'error': '分类不存在'
        }, status=400)
    except Exception as e:
        return JsonResponse({
            'error': f'创建失败: {str(e)}'
        }, status=500)

from django.core.files.storage import default_storage
from django.core.files.base import ContentFile

def file_upload_view(request):
    """文件上传视图"""
    
    if request.method == 'POST':
        # 检查是否有文件上传
        if 'file' in request.FILES:
            uploaded_file = request.FILES['file']
            
            # 文件信息
            file_name = uploaded_file.name
            file_size = uploaded_file.size
            file_content_type = uploaded_file.content_type
            
            # 验证文件类型
            allowed_types = ['image/jpeg', 'image/png', 'image/gif']
            if file_content_type not in allowed_types:
                return JsonResponse({
                    'error': '不支持的文件类型'
                }, status=400)
            
            # 验证文件大小（5MB限制）
            if file_size > 5 * 1024 * 1024:
                return JsonResponse({
                    'error': '文件大小不能超过5MB'
                }, status=400)
            
            try:
                # 保存文件
                file_path = default_storage.save(
                    f'uploads/{file_name}', 
                    ContentFile(uploaded_file.read())
                )
                
                return JsonResponse({
                    'success': True,
                    'file_path': file_path,
                    'file_name': file_name,
                    'file_size': file_size
                })
                
            except Exception as e:
                return JsonResponse({
                    'error': f'文件保存失败: {str(e)}'
                }, status=500)
        else:
            return JsonResponse({
                'error': '没有文件上传'
            }, status=400)
    
    return render(request, 'file_upload.html')
```

def multiple_file_upload_view(request):
    """多文件上传视图"""
    
    if request.method == 'POST':
        uploaded_files = request.FILES.getlist('files')
        
        if not uploaded_files:
            return JsonResponse({
                'error': '没有文件上传'
            }, status=400)
        
        results = []
        
        for uploaded_file in uploaded_files:
            try:
                # 生成唯一文件名
                import uuid
                file_extension = uploaded_file.name.split('.')[-1]
                unique_filename = f"{uuid.uuid4()}.{file_extension}"
                
                # 保存文件
                file_path = default_storage.save(
                    f'uploads/{unique_filename}', 
                    ContentFile(uploaded_file.read())
                )
                
                results.append({
                    'original_name': uploaded_file.name,
                    'saved_path': file_path,
                    'size': uploaded_file.size,
                    'status': 'success'
                })
                
            except Exception as e:
                results.append({
                    'original_name': uploaded_file.name,
                    'status': 'error',
                    'error': str(e)
                })
        
        return JsonResponse({
            'success': True,
            'results': results
        })
    
    return render(request, 'multiple_file_upload.html')

                    # 普通页面请求
                    context = {
                        'article': article,
                        'request_info': request_info
                    }
                    return render(request, 'article_detail.html', context)
                    
            except Article.DoesNotExist:
                if request.is_ajax():
                    return JsonResponse({'error': '文章不存在'}, status=404)
                else:
                    return HttpResponse("文章不存在", status=404)
        else:
            # 获取文章列表
            articles = Article.objects.all()
            
            # 处理查询参数
            category = request.GET.get('category')
            if category:
                articles = articles.filter(category__slug=category)
            
            search = request.GET.get('search')
            if search:
                articles = articles.filter(title__icontains=search)
            
            # 排序
            sort_by = request.GET.get('sort', 'created_at')
            if sort_by in ['title', 'created_at', 'views']:
                if request.GET.get('order') == 'desc':
                    articles = articles.order_by(f'-{sort_by}')
                else:
                    articles = articles.order_by(sort_by)
            
            # 分页
            page = int(request.GET.get('page', 1))
            per_page = int(request.GET.get('per_page', 10))
            
            start = (page - 1) * per_page
            end = start + per_page
            
            articles_page = articles[start:end]
            
            if request.is_ajax():
                data = []
                for article in articles_page:
                    data.append({
                        'id': article.id,
                        'title': article.title,
                        'author': article.author.username,
                        'views': article.views,
                        'created_at': article.created_at.isoformat(),
                    })
                
                return JsonResponse({
                    'articles': data,
                    'page': page,
                    'per_page': per_page,
                    'total': articles.count(),
                    'request_info': request_info
                })
            else:
                context = {
                    'articles': articles_page,
                    'request_info': request_info,
                    'categories': Category.objects.all()
                }
                return render(request, 'article_list.html', context)
    
    elif request.method == "POST":
        # 创建文章
        if request.content_type == 'application/json':
            try:
                data = json.loads(request.body)
            except json.JSONDecodeError:
                return JsonResponse({'error': '无效的JSON格式'}, status=400)
        else:
            data = request.POST
        
        # 验证数据
        title = data.get('title')
        content = data.get('content')
        category_id = data.get('category')
        
        if not all([title, content, category_id]):
            return JsonResponse({
                'error': '请填写所有必填字段',
                'missing': [k for k, v in {'title': title, 'content': content, 'category': category_id}.items() if not v]
            }, status=400)
        
        try:
            category = Category.objects.get(pk=category_id)
            article = Article.objects.create(
                title=title,
                content=content,
                category=category,
                author=request.user
            )
            
            return JsonResponse({
                'success': True,
                'article_id': article.id,
                'message': '文章创建成功'
            }, status=201)
            
        except Category.DoesNotExist:
            return JsonResponse({'error': '分类不存在'}, status=400)
        except Exception as e:
            return JsonResponse({'error': f'创建失败: {str(e)}'}, status=500)
    
    elif request.method == "PUT":
        # 更新文章
        if not article_id:
            return JsonResponse({'error': '缺少文章ID'}, status=400)
        
        try:
            article = get_object_or_404(Article, pk=article_id)
            
            if request.content_type == 'application/json':
                data = json.loads(request.body)
            else:
                data = request.POST
            
            # 更新字段
            if 'title' in data:
                article.title = data['title']
            if 'content' in data:
                article.content = data['content']
            if 'category' in data:
                try:
                    category = Category.objects.get(pk=data['category'])
                    article.category = category
                except Category.DoesNotExist:
                    return JsonResponse({'error': '分类不存在'}, status=400)
            
            article.save()
            
            return JsonResponse({'success': True, 'message': '文章更新成功'})
            
        except json.JSONDecodeError:
            return JsonResponse({'error': '无效的JSON格式'}, status=400)
        except Exception as e:
            return JsonResponse({'error': f'更新失败: {str(e)}'}, status=500)
    
    elif request.method == "DELETE":
        # 删除文章
        if not article_id:
            return JsonResponse({'error': '缺少文章ID'}, status=400)
        
        try:
            article = get_object_or_404(Article, pk=article_id)
            article.delete()
            
            return JsonResponse({'success': True, 'message': '文章删除成功'})
            
        except Exception as e:
            return JsonResponse({'error': f'删除失败: {str(e)}'}, status=500)
    
    else:
        return JsonResponse({'error': '不支持的HTTP方法'}, status=405)
```
